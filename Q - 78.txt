class Solution {
    public List<List<Integer>> subsets(int[] nums) {
        List<List<Integer>> reslist = new ArrayList<>();

        bt(reslist , nums , new ArrayList<>() , 0);

        return reslist;
    }

    public void bt(List<List<Integer>> reslist , int[] nums , List<Integer> temp , int start){

            reslist.add(new ArrayList<>(temp));

            for(int i = start ; i < nums.length ; i++){
                temp.add(nums[i]);

                bt(reslist , nums , temp , i+1);

                temp.remove(temp.size() - 1);
            }
    }
}
